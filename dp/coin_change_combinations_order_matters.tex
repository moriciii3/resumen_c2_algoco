\subsubsection{Combinaciones de Monedas (El Orden Importa)}

Busca el número total de formas de formar el monto objetivo $X$, donde la secuencia de monedas usadas sí importa. La estrategia es iterar sobre los montos (exterior) y luego sobre las monedas (interior).

\paragraph{Fórmula DP:}
$$
DP[i] = \sum_{j} DP[i - C_j] \quad \text{si } i \ge C_j
$$

\paragraph{Complejidad:}
* \textbf{Tiempo:} $O(N \cdot X)$.
* \textbf{Espacio:} $O(X)$.

\subsubsection{Código C++ (Combinaciones - Orden Importa)}
\begin{lstlisting}[
	caption={Coin Change: Combinaciones donde el orden importa (DP)},
	extendedchars=true, 
	literate={ñ}{{\~n}}1 {Ñ}{{\~N}}1 {á}{{\'a}}1 {é}{{\'e}}1 {í}{{\'i}}1 {ó}{{\'o}}1 {ú}{{\'u}}1 {Á}{{\'A}}1 {É}{{\'E}}1 {Í}{{\'I}}1 {Ó}{{\'O}}1 {Ú}{{\'U}}1
	]
	#include <bits/stdc++.h>
	using 11 long long;
	using namespace std;
	
	#define mod 1000000007
	#define endl "\n"
	
	int main() {
		11 n, x;
		cin >> n >> x;
		
		11 dp [x + 1] = {0}; 
		11 coins [n];
		
		for (11 i=0; i < n; i++) {
			cin >> coins [i];
		}
		
		dp [0] = 1;
		
		for (11 i=1; i <= x; i++) { 
			for (11 j=0; j<n; j++) {
				if (i >= coins[j]) {
					dp[i] = (dp[i] + dp[i - coins [j]]) % mod;
				}
			}
		}
		
		cout << dp[x] << endl;
		return 0;
	}
\end{lstlisting}
